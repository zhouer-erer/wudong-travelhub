flowchart TD
    A[开始] --> B[用户输入账号密码]
    B --> C{查询管理员信息}
    C -->|用户名不存在| D[返回401错误]
    D --> Z[结束]
    C -->|用户名存在| E{检查账号状态}
    E -->|账号已禁用| F[返回403错误]
    F --> Z
    E -->|账号被锁定| G[返回423锁定提示]
    G --> Z
    E -->|账号正常| H{验证密码}
    H -->|密码错误| I{失败次数>=5?}
    I -->|是| J[锁定账号30分钟]
    J --> K[返回423锁定提示]
    K --> Z
    I -->|否| L[累计失败次数]
    L --> M[返回401错误提示]
    M --> Z
    H -->|密码正确| N[清除锁定状态]
    N --> O[生成6位验证码]
    O --> P[生成UUID临时令牌]
    P --> Q[存储到内存缓存]
    Q --> R[返回手机号掩码和令牌]
    R --> S[用户输入验证码]
    S --> T{验证验证码}
    T -->|无效/过期| U[返回401错误]
    U --> Z
    T -->|验证通过| V[生成JWT Token]
    V --> W{检测异地登录}
    W -->|是| X[发送系统消息提醒]
    X --> Y[更新登录信息]
    W -->|否| Y
    Y --> AA[返回Token和用户信息]
    AA --> Z[结束]

flowchart TD
    A[开始] --> B[用户发起API请求]
    B --> C[携带JWT Token]
    C --> D{验证Token}
    D -->|无效/过期| E[返回401错误]
    E --> Z[结束]
    D -->|验证通过| F{解析用户角色}
    F -->|商家角色| G{检查接口权限}
    G -->|商家接口| H[设置商家上下文]
    H --> I[放行请求]
    G -->|管理后台接口| J[返回403错误]
    J --> Z
    F -->|管理员角色| K[查询管理员信息]
    K --> L{检查账号状态}
    L -->|账号禁用| M[返回401错误]
    M --> Z
    L -->|账号正常| N[设置管理员上下文]
    N --> O{路由在白名单?}
    O -->|是| I
    O -->|否| P[匹配路由权限码]
    P --> Q{路由在映射表?}
    Q -->|否| I
    Q -->|是| R{查询角色权限}
    R -->|Redis缓存命中| S[获取权限列表]
    R -->|Redis未命中| T[数据库查询权限]
    T --> U[缓存权限到Redis]
    U --> S
    S --> V{权限校验通过?}
    V -->|否| W[返回403错误]
    W --> Z
    V -->|是| I
    I --> X[执行业务逻辑]
    X --> Y[返回响应结果]
    Y --> Z[结束]

    A[开始] --> B[用户提交入驻申请]

    A[开始] --> B[订单完成]
