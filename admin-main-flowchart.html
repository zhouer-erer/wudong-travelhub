flowchart TD
    subgraph 登录认证 ["登录认证模块"]
        A[开始] --> B[输入账号密码]
        B --> C{验证账号}
        C -->|失败| D[返回错误]
        C -->|成功| E[发送验证码]
        E --> F[验证验证码]
        F -->|失败| D
        F -->|成功| G[生成JWT Token]
        G --> H[记录登录信息]
        H --> I[进入管理后台]
    end

    subgraph 权限管理 ["权限管理模块"]
        J[验证Token] --> K{Token有效?}
        K -->|否| L[拒绝访问]
        K -->|是| M[解析角色权限]
        M --> N{权限校验}
        N -->|不足| L
        N -->|通过| O[放行请求]
        O --> P[执行业务逻辑]
    end

    subgraph 商家管理 ["商家管理模块"]
        Q[提交入驻申请] --> R[审核申请]
        R -->|驳回| S[通知申请人]
        R -->|通过| T[创建商家账号]
        T --> U[商家登录]
        U --> V[完善店铺信息]
        V --> W[发布商品]
        W --> X{违规检查}
        X -->|违规| Y[封禁店铺]
        X -->|正常| Z[继续运营]
    end

    subgraph 订单管理 ["订单管理模块"]
        AA[用户下单] --> AB[生成订单]
        AB --> AC[用户支付]
        AC --> AD[商家发货]
        AD --> AE[订单完成]
        AE --> AF{申请退款?}
        AF -->|否| AG[订单结束]
        AF -->|是| AH[审核退款]
        AH -->|拒绝| AI[通知用户]
        AH -->|通过| AJ[执行退款]
        AJ --> AK[更新订单状态]
    end

    subgraph 财务结算 ["财务结算模块"]
        AL[订单完成] --> AM[生成财务记录]
        AM --> AN[计算抽佣]
        AN --> AO[等待结算周期]
        AO --> AP[自动结算]
        AP --> AQ[生成结算单]
        AQ --> AR[商家申请提现]
        AR --> AS[审核提现]
        AS -->|拒绝| AT[通知商家]
        AS -->|通过| AU[执行转账]
        AU --> AV[生成财务报表]
    end

    subgraph 内容运营 ["内容运营模块"]
        AW[发布公告] --> AX[管理轮播图]
        AX --> AY[管理活动横幅]
        AY --> AZ[配置推荐位]
        AZ --> BA[内容审核]
    end

    subgraph 系统管理 ["系统管理模块"]
        BB[系统配置] --> BC[敏感词管理]
        BC --> BD[操作日志]
        BD --> BE[消息模板]
    end
